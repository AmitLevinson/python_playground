from urllib.request import urlopen
import json
import pandas as pd
import time

TOP_API_URL = 'https://wikimedia.org/api/rest_v1/metrics/pageviews/top/he.wikipedia/all-access/{year}/{month}/{day}'

def get_traffic(year, month, day):
    url = TOP_API_URL.format(year=year,
                             month=month,
                             day=day)
    resp = urlopen(url)
    resp_bytes = resp.read()
    data = json.loads(resp_bytes)
    articles = data['items'][0]['articles']
    return articles
dates_range = pd.date_range('2023-01-01','2023-01-03',freq='d')

lst = []

for i, date in enumerate(dates_range):
    print(date)
    df = get_traffic(f'{dates_range[i]:%Y}', f"{dates_range[i]:%m}", f"{dates_range[i]:%d}") # extracting year, month and day
    df = pd.DataFrame(df)
    df['date'] = date # Add the current data collected
    lst.append(df) # appending the new df to the previous ones collected
    time.sleep(10)
    
wiki_data = pd.concat(lst)
